# 8 golden USCRN pairs used
import requests
import tarfile
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.gridspec as gridspec # For custom subplot layout
import matplotlib.patches as patches # For the highlight box
import matplotlib.dates as mdates   # For date formatting
import statsmodels.api as sm
import io
from calendar import month_abbr # For month names

# === Settings ===
baseline_start = 1920 # Start year for baseline calculation
baseline_end = 1940   # End year for baseline calculation
min_months_annual = 11 # Minimum number of valid months to calculate annual average
start_year = 1910     # First year of data to consider
end_year = 2024       # Last year of data to consider
# smoothing_frac_annual = 0.13 # No longer needed for plot 1
trend_start_year = 1910 # Year to start the long-term linear trend calculation
monthly_plot_start_year = 2005 # Start year for the detailed monthly plot
monthly_breakpoint_year = 2015 # Year *after* which the second trend starts (i.e., >= 2015)
# Breakpoint explanation focuses around mid-2015 visually
smoothing_frac_monthly = 0.15 # Smoother LOESS for monthly data (adjust as needed)

# === Step 1: Download and extract data ===
url = "https://www.ncei.noaa.gov/pub/data/ghcn/v4/ghcnm.tavg.latest.qcu.tar.gz"
print(f"Downloading data from {url}...")
try:
    response = requests.get(url, stream=True, timeout=60)
    response.raise_for_status()
    tar_bytes = io.BytesIO(response.content)
    print("Download complete.")
except requests.exceptions.RequestException as e:
    print(f"Error downloading data: {e}")
    exit()

# Station pairs with names
pairs_with_names = [
    ("USC00012813", "USW00063869", "Demopolis, AL"),
    ("USC00294426", "USW00003074", "Jefferson City, MO"),
    ("USC00402202", "USW00063855", "Crossville, TN"),
    ("USC00250030", "USW00094077", "Abilene, TX"),
    ("USC00018385", "USW00073801", "Tuscaloosa, AL"),
    ("USC00380764", "USW00063826", "Fort Wayne, IN"),
    ("USC00013160", "USW00063892", "Enterprise, AL"),
    ("USC00348501", "USW00053926", "Spearfish, SD")
]
pairs = [(p[0], p[1]) for p in pairs_with_names]
station_ids = {sid for p in pairs for sid in p}
station_to_pair = {sid: pair for pair in pairs for sid in pair}
print(f"Using {len(pairs)} station pairs.")

# === Step 2: Extract relevant lines from the archive ===
filtered_lines = []
print("Extracting and filtering data...")
try:
    with tarfile.open(fileobj=tar_bytes, mode="r:gz") as tar:
        dat_member = next((m for m in tar.getmembers() if m.name.endswith(".qcu.dat")), None)
        if dat_member is None: print("Error: Could not find .qcu.dat file."); exit()
        dat_file = tar.extractfile(dat_member)
        if dat_file is None: print("Error: Could not extract the .qcu.dat file."); exit()

        for line_bytes in dat_file:
            line = line_bytes.decode("utf-8")
            sid = line[0:11].strip()
            if sid in station_ids:
                element = line[15:19]
                if element == "TAVG":
                    filtered_lines.append(line)
        print(f"Found {len(filtered_lines)} relevant TAVG records.")

except tarfile.TarError as e: print(f"Error reading tar file: {e}"); exit()
except Exception as e: print(f"An unexpected error during extraction: {e}"); exit()

# === Step 3: Parse *MONTHLY* temperature data ===
monthly_records = []
print("Parsing monthly temperature data...")
for line in filtered_lines:
    try:
        sid = line[0:11].strip()
        year = int(line[11:15])
        if year < start_year or year > end_year: continue

        for m in range(12):
            month = m + 1
            start_col = 19 + m * 8
            end_col = start_col + 5
            value_str = line[start_col:end_col].strip()
            value = int(value_str)
            temp = value / 100.0 if value != -9999 else np.nan
            monthly_records.append([year, month, temp, sid])
    except ValueError as e: print(f"Warning: Skipping line due to parsing error: {line.strip()} ({e})")
    except IndexError as e: print(f"Warning: Skipping line due to index error: {line.strip()} ({e})")

df_monthly_raw = pd.DataFrame(monthly_records, columns=["year", "month", "tavg", "station_id"])
df_monthly_raw['pair_id'] = df_monthly_raw['station_id'].map(station_to_pair)
print(f"Parsed {len(df_monthly_raw)} monthly temperature records.")

if df_monthly_raw.empty: print("Error: No valid monthly data parsed. Exiting."); exit()

# === Step 4: Calculate Pairwise MONTHLY Baselines and Anomalies ===
all_pairs_monthly_anomalies = []
print(f"Calculating monthly anomalies relative to {baseline_start}-{baseline_end} baseline (pairwise)...")
for pair_id in pairs:
    legacy_id, uscrn_id = pair_id
    df_pair_monthly = df_monthly_raw[df_monthly_raw['pair_id'] == pair_id].copy()
    df_pair_monthly_agg = df_pair_monthly.groupby(['year', 'month'])['tavg'].mean().reset_index()
    baseline_data = df_pair_monthly_agg[
        (df_pair_monthly_agg["year"] >= baseline_start) &
        (df_pair_monthly_agg["year"] <= baseline_end)
    ]
    pair_monthly_baselines = baseline_data.groupby('month')['tavg'].mean()

    if len(pair_monthly_baselines) == 12:
        df_pair_monthly_agg['baseline'] = df_pair_monthly_agg['month'].map(pair_monthly_baselines)
        df_pair_monthly_agg['monthly_anomaly'] = df_pair_monthly_agg['tavg'] - df_pair_monthly_agg['baseline']
        all_pairs_monthly_anomalies.append(df_pair_monthly_agg[['year', 'month', 'monthly_anomaly']])
    else:
        print(f"  Warning: Pair {pair_id} missing baseline data for some months in {baseline_start}-{baseline_end}. Skipping pair.")

if not all_pairs_monthly_anomalies:
    print("Error: No pairs had sufficient baseline data for all months. Cannot proceed."); exit()

# === Step 5: Aggregate MONTHLY anomalies & derive Annual ===
df_all_monthly = pd.concat(all_pairs_monthly_anomalies)
df_agg_monthly = df_all_monthly.groupby(['year', 'month'])['monthly_anomaly'].mean().reset_index()

annual_counts = df_agg_monthly.groupby('year')['monthly_anomaly'].count()
valid_years = annual_counts[annual_counts >= min_months_annual].index
df_agg_annual = df_agg_monthly[df_agg_monthly['year'].isin(valid_years)].groupby('year')['monthly_anomaly'].mean().reset_index()
df_agg_annual.rename(columns={'monthly_anomaly': 'anomaly'}, inplace=True)
print(f"Aggregated annual anomalies for {len(df_agg_annual)} years.")

# === Step 6: Calculate Annual Linear Trend ===
print(f"Calculating annual linear trend from {trend_start_year}...")
df_trend_annual = df_agg_annual[df_agg_annual['year'] >= trend_start_year].copy()
slope_per_century_annual = None
trend_end_year_annual = None
if len(df_trend_annual) > 2: # Need at least 3 points for trend
    X_annual = sm.add_constant(df_trend_annual['year'])
    y_annual = df_trend_annual['anomaly']
    model_annual = sm.OLS(y_annual, X_annual).fit()
    slope_per_year_annual = model_annual.params['year']
    slope_per_century_annual = slope_per_year_annual * 100
    trend_end_year_annual = int(df_trend_annual['year'].max())
    df_trend_annual['trend_fit'] = model_annual.predict(X_annual)
    print(f"Annual Linear trend ({trend_start_year}–{trend_end_year_annual}): {slope_per_century_annual:+.2f} °C per Century")
else:
     print(f"Warning: Not enough annual data points from {trend_start_year} onwards to calculate trend.")


# === Step 7: Prepare Monthly Data & Trends (2005 onwards) ===
df_monthly_focus = df_agg_monthly[df_agg_monthly['year'] >= monthly_plot_start_year].copy()

if df_monthly_focus.empty:
    print(f"No monthly data available from {monthly_plot_start_year} onwards. Cannot generate monthly plots.")
    # Set flags/empty data structures if needed, or exit if essential
    monthly_data_available = False
else:
    monthly_data_available = True
    df_monthly_focus['date'] = pd.to_datetime(df_monthly_focus[['year', 'month']].assign(DAY=1))
    df_monthly_focus = df_monthly_focus.sort_values(by='date').set_index('date')
    df_monthly_focus['time_float'] = df_monthly_focus['year'] + (df_monthly_focus['month'] - 1) / 12.0

    # --- Monthly LOESS (for Plot 2) ---
    print(f"Applying LOESS smoothing (frac={smoothing_frac_monthly}) to monthly data ({monthly_plot_start_year}-)...")
    df_monthly_loess = df_monthly_focus.dropna(subset=['monthly_anomaly', 'time_float'])
    if not df_monthly_loess.empty:
        loess_monthly = sm.nonparametric.lowess(endog=df_monthly_loess["monthly_anomaly"],
                                                exog=df_monthly_loess['time_float'],
                                                frac=smoothing_frac_monthly)
        loess_series = pd.Series(loess_monthly[:, 1], index=df_monthly_loess.index)
    else:
        loess_series = pd.Series(dtype=float)
        print("Warning: No valid data for monthly LOESS smoothing.")

    # --- Single Monthly Linear Trend (for Plot 2) ---
    print(f"Calculating single monthly linear trend from {monthly_plot_start_year}...")
    df_monthly_ols = df_monthly_focus.dropna(subset=['monthly_anomaly', 'time_float'])
    slope_per_decade_monthly_full = None
    trend_end_year_monthly = None
    if len(df_monthly_ols) > 2:
        X_monthly_full = sm.add_constant(df_monthly_ols['time_float'])
        y_monthly_full = df_monthly_ols['monthly_anomaly']
        model_monthly_full = sm.OLS(y_monthly_full, X_monthly_full).fit()
        slope_per_decade_monthly_full = model_monthly_full.params['time_float'] * 10
        df_monthly_focus['trend_fit_full'] = model_monthly_full.predict(sm.add_constant(df_monthly_focus['time_float'], has_constant='add'))
        trend_end_year_monthly = df_monthly_focus['year'].max()
        trend_end_month_monthly = df_monthly_focus[df_monthly_focus['year'] == trend_end_year_monthly]['month'].max()
        print(f"Single Monthly Linear trend ({monthly_plot_start_year}-Jan to {trend_end_year_monthly}-{month_abbr[trend_end_month_monthly]}): {slope_per_decade_monthly_full:+.2f} °C per Decade")
    else:
        print("Warning: Not enough valid data points for overall monthly trend calculation.")

    # --- Breakpoint Trends (for Plot 3) ---
    print(f"Calculating breakpoint trends around {monthly_breakpoint_year}...")
    df_monthly_pre_break = df_monthly_focus[df_monthly_focus['year'] < monthly_breakpoint_year].copy()
    df_monthly_post_break = df_monthly_focus[df_monthly_focus['year'] >= monthly_breakpoint_year].copy()
    slope_per_decade_pre = None
    slope_per_decade_post = None

    df_ols_pre = df_monthly_pre_break.dropna(subset=['monthly_anomaly', 'time_float'])
    if len(df_ols_pre) > 2:
        y_pre = df_ols_pre['monthly_anomaly']
        X_pre = sm.add_constant(df_ols_pre['time_float'])
        model_pre = sm.OLS(y_pre, X_pre).fit()
        slope_per_decade_pre = model_pre.params['time_float'] * 10
        df_monthly_pre_break['trend_fit'] = model_pre.predict(sm.add_constant(df_monthly_pre_break['time_float'], has_constant='add'))
        print(f"  Pre-{monthly_breakpoint_year} trend ({monthly_plot_start_year}–{monthly_breakpoint_year-1}): {slope_per_decade_pre:+.2f} °C / Decade")
    else:
        print(f"  Not enough valid data points pre-{monthly_breakpoint_year} to calculate trend.")

    df_ols_post = df_monthly_post_break.dropna(subset=['monthly_anomaly', 'time_float'])
    if len(df_ols_post) > 2:
        y_post = df_ols_post['monthly_anomaly']
        X_post = sm.add_constant(df_ols_post['time_float'])
        model_post = sm.OLS(y_post, X_post).fit()
        slope_per_decade_post = model_post.params['time_float'] * 10
        df_monthly_post_break['trend_fit'] = model_post.predict(sm.add_constant(df_monthly_post_break['time_float'], has_constant='add'))
        print(f"  Post-{monthly_breakpoint_year} trend ({monthly_breakpoint_year}–{trend_end_year_monthly}): {slope_per_decade_post:+.2f} °C / Decade")
    else:
        print(f"  Not enough valid data points post-{monthly_breakpoint_year} to calculate trend.")

# === Step 8: Create Figure and Subplot Grid ===
print("Generating plots...")
plt.style.use('dark_background')
fig = plt.figure(figsize=(14, 10)) # Adjust overall figure size as needed

# Define grid: 2 rows, 2 columns. Make top row taller.
gs = gridspec.GridSpec(2, 2, height_ratios=[2, 1.3], width_ratios=[1, 1])

ax1 = fig.add_subplot(gs[0, :]) # Top plot spans both columns
ax2 = fig.add_subplot(gs[1, 0]) # Bottom left
ax3 = fig.add_subplot(gs[1, 1]) # Bottom right

# === Step 9: Plot 1 (Top: Annual Data + Thick Linear Trend) ===
ax1.plot(df_agg_annual["year"], df_agg_annual["anomaly"], color='cyan',
         linewidth=1.8, marker='o', markersize=3.5, alpha=0.7, label="Annual Anomaly")

if slope_per_century_annual is not None and 'trend_fit' in df_trend_annual.columns:
    ax1.plot(df_trend_annual['year'], df_trend_annual['trend_fit'], color='cyan', linewidth=4.0, alpha=0.9, # Thick Cyan Trend
             label=f"Linear Trend ({trend_start_year}–{trend_end_year_annual})")
    text_x = df_trend_annual['year'].mean()
    # Adjust y-position if needed based on ylim
    current_ylim = ax1.get_ylim()
    text_y = df_trend_annual['trend_fit'].iloc[-1] + 0.05 * (current_ylim[1] - current_ylim[0])
    trend_text = f"{slope_per_century_annual:+.2f} °C / Century"
    ax1.text(text_x, text_y, trend_text, color='cyan', fontsize=12, weight='bold', ha='center') # Cyan Text

ax1.axhline(0, color='white', linestyle='--', linewidth=0.7, alpha=0.5)

# Add Highlight Box
highlight_start = monthly_plot_start_year
highlight_end = end_year
actual_max_year = df_agg_annual['year'].max()
if highlight_end > actual_max_year: highlight_end = actual_max_year
box_ymin, box_ymax = (-2, 1) # Set fixed y limits for consistency
ax1.set_ylim(box_ymin, box_ymax)

rect = patches.Rectangle((highlight_start - 0.5, box_ymin),
                         highlight_end - highlight_start + 1, box_ymax - box_ymin,
                         linewidth=1, edgecolor='none', facecolor='grey', alpha=0.25, zorder=0,
                         label=f"Focus Period ({highlight_start}-{int(highlight_end)})")
ax1.add_patch(rect)

# Customize Plot 1
ax1.set_title(f"US Temperature Anomalies: 8 USCRN Pairs (Baseline {baseline_start}–{baseline_end})", fontsize=15, weight='bold', pad=10)
# ax1.set_xlabel("Year", fontsize=12) # Remove xlabel for top plot to save space
ax1.tick_params(axis='x', labelbottom=False) # Hide x-axis labels
ax1.set_ylabel("Temp. Anomaly (°C)", fontsize=12)
ax1.grid(True, linestyle=':', linewidth=0.5, color='gray', alpha=0.6)
ax1.legend(loc="upper left", fontsize=10)


# === Step 10: Plot 2 (Bottom Left: Monthly Focus, LOESS + Single Trend) ===
if monthly_data_available:
    ax2.plot(df_monthly_focus.index, df_monthly_focus['monthly_anomaly'], label="Monthly Anomaly", color='cyan',
             linewidth=0.6, alpha=0.7)

    if not loess_series.empty:
        ax2.plot(loess_series.index, loess_series.values, color='yellow', linewidth=2.0, alpha=0.9,
                 label=f"LOESS (frac={smoothing_frac_monthly})")

    if 'trend_fit_full' in df_monthly_focus.columns and slope_per_decade_monthly_full is not None:
        ax2.plot(df_monthly_focus.index, df_monthly_focus['trend_fit_full'], color='magenta', linewidth=2.5, alpha=0.9,
                 label="Overall Linear Trend")
        text_x_monthly_date = df_monthly_focus.index[len(df_monthly_focus) // 2]
        text_y_monthly = df_monthly_focus['trend_fit_full'].quantile(0.8) + 0.1
        trend_text_monthly = f"{slope_per_decade_monthly_full:+.2f} °C / Decade"

    ax2.axhline(0, color='white', linestyle='--', linewidth=0.7, alpha=0.5)

    # Customize Plot 2
    ax2.set_title(f"Monthly Detail ({monthly_plot_start_year}–{trend_end_year_monthly}): Overall Trend", fontsize=12, weight='bold', pad=8)
    ax2.set_xlabel("Date", fontsize=11)
    ax2.set_ylabel("Monthly Anomaly (°C)", fontsize=11)
    # Determine Y limits based on this plot's data, possibly constrained
    min_anom, max_anom = df_monthly_focus['monthly_anomaly'].quantile([0.01, 0.99])
    if pd.isna(min_anom): min_anom = df_monthly_focus['monthly_anomaly'].min()
    if pd.isna(max_anom): max_anom = df_monthly_focus['monthly_anomaly'].max()

    ax2.set_ylim(current_ylim_ax2)

    ax2.legend(loc="upper left", fontsize=9)
    ax2.xaxis.set_major_locator(mdates.YearLocator(3)) # Adjust tick frequency
    ax2.xaxis.set_major_formatter(mdates.DateFormatter('%Y'))
else:
    ax2.text(0.5, 0.5, "Monthly data not available\nin this period", ha='center', va='center', fontsize=12, color='gray')
    ax2.set_xticks([])
    ax2.set_yticks([])

if monthly_data_available:

else:
