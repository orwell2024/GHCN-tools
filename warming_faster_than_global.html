def download_file(url, target_path, file_type="Data"):
    print(f"Downloading {file_type} from {url} to {target_path}...")
    try:
        headers = {'User-Agent': 'Mozilla/5.0'}
        with requests.get(url, stream=True, timeout=120, headers=headers) as r:
            r.raise_for_status()
            total_size = int(r.headers.get('content-length', 0))
            block_size = 8192
            start_time = time.time(); last_print_time = start_time; bytes_downloaded = 0
            with open(target_path, 'wb') as f:
                for chunk in r.iter_content(chunk_size=block_size):
                    f.write(chunk); bytes_downloaded += len(chunk)
                    current_time = time.time()
                    if total_size > 0:
                         done = int(50 * bytes_downloaded / total_size)
                         elapsed_time = current_time - start_time
                         speed = bytes_downloaded / elapsed_time / 1024 if elapsed_time > 0 else 0
                         if bytes_downloaded == total_size or current_time - last_print_time > 0.5 or bytes_downloaded == block_size:
                              print(f"\r[{'=' * done}{' ' * (50-done)}] {bytes_downloaded/1024/1024:.1f}/{total_size/1024/1024:.1f} MB ({speed:.1f} KiB/s) ", end='')
                              last_print_time = current_time
            if total_size > 0 : print()
            print(f"{file_type} Download complete.")
        return True
    except requests.exceptions.RequestException as e:
        print(f"\nError downloading {url}: {e}")
        return False
    except Exception as e:
        print(f"\nAn unexpected error occurred during {file_type} download: {e}")
        return False

def get_data_path(url, local_filename, cache_dir, file_type_desc, force_download=False):
    os.makedirs(cache_dir, exist_ok=True)
    local_filepath = os.path.join(cache_dir, local_filename)
    use_this_path = None; download_needed = False
    if force_download: download_needed = True
    elif use_local_data_if_available and os.path.exists(local_filepath):
        try:
            file_age_seconds = time.time() - os.path.getmtime(local_filepath)
            if file_age_seconds < 86400: use_this_path = local_filepath
            else: download_needed = True
        except Exception as e: download_needed = True
    else: download_needed = True
    if download_needed:
        print(f"Attempting download for {file_type_desc}...")
        if download_file(url, local_filepath, file_type_desc): use_this_path = local_filepath
        elif os.path.exists(local_filepath) and use_local_data_if_available:
             print(f"Download failed. Falling back to existing local {file_type_desc} file.")
             use_this_path = local_filepath
        else: print(f"Error: {file_type_desc} download failed, no local file."); use_this_path = None
    else: print(f"Using existing local {file_type_desc} file: {local_filepath}")
    return use_this_path

# --- Helper Function for Anomaly Calculation ---
def calculate_anomalies(df_data, station_ids, baseline_start, baseline_end, series_name, station_metadata=None, grid_res=None, lat_min=-90, lat_max=90, lon_min=-180, lon_max=180, do_gridding=False):
    print(f"\nCalculating anomalies for: {series_name} (Baseline: {baseline_start}-{baseline_end})")
    station_ids_set = set(station_ids)
    if not station_ids_set:
        print(f"  Warning: No station IDs provided for {series_name}. Returning empty series.")
        return pd.Series(index=all_years_index, dtype=float)
    df_calc = df_data[df_data['station_id'].isin(station_ids_set)].copy()
    if df_calc.empty:
        print(f"  Warning: No data found for stations in {series_name}.")
        return pd.Series(index=all_years_index, dtype=float)
    baseline_data = df_calc[(df_calc['year'] >= baseline_start) & (df_calc['year'] <= baseline_end)]
    station_baselines = baseline_data.groupby('station_id')['tavg'].mean()
    stations_with_baseline = set(station_baselines.index)
    num_baseline = len(stations_with_baseline)
    print(f"  Found {num_baseline} of {len(station_ids_set)} stations with baseline data for {series_name}.")
    if num_baseline == 0 and not df_calc.empty:
         print(f"  Warning: No baseline data found for {series_name}, cannot calculate anomalies.")
         return pd.Series(index=all_years_index, dtype=float)
    df_anom_calc = df_calc[df_calc['station_id'].isin(stations_with_baseline)].copy()
    if df_anom_calc.empty:
        print(f"  Warning: No data remaining after baseline filter for {series_name}.")
        return pd.Series(index=all_years_index, dtype=float)
    df_anom_calc['baseline_mean'] = df_anom_calc['station_id'].map(station_baselines)
    df_anom_calc['anomaly'] = df_anom_calc['tavg'] - df_anom_calc['baseline_mean']
    print(f"  Calculated {len(df_anom_calc)} station-level anomalies for {series_name} (using {num_baseline} baselines).")
    if not do_gridding:
        anomaly_series = df_anom_calc.groupby('year')['anomaly'].mean().reindex(all_years_index)
        print(f"  Finished simple average anomaly for {series_name}.")
        return anomaly_series
    else: # Gridding logic
        if station_metadata is None or grid_res is None: print(" Error: Gridding needs metadata/res."); return pd.Series(index=all_years_index, dtype=float)
        df_anom_calc_grid = df_anom_calc.merge(station_metadata[[meta_lat_col, meta_lon_col]], left_on='station_id', right_index=True, how='left')
        df_anom_calc_grid.dropna(subset=[meta_lat_col, meta_lon_col], inplace=True)
        if df_anom_calc_grid.empty: print(f"  Warn: No coords for gridding {series_name}"); return pd.Series(index=all_years_index, dtype=float)
        df_anom_calc_grid = df_anom_calc_grid[
            (df_anom_calc_grid[meta_lat_col] >= lat_min) & (df_anom_calc_grid[meta_lat_col] <= lat_max) &
            (df_anom_calc_grid[meta_lon_col] >= lon_min) & (df_anom_calc_grid[meta_lon_col] <= lon_max)]
        if df_anom_calc_grid.empty: print(f"  Warn: No stations in grid bounds for {series_name}"); return pd.Series(index=all_years_index, dtype=float)
        df_anom_calc_grid[meta_lat_col] = pd.to_numeric(df_anom_calc_grid[meta_lat_col], errors='coerce'); df_anom_calc_grid[meta_lon_col] = pd.to_numeric(df_anom_calc_grid[meta_lon_col], errors='coerce')
        df_anom_calc_grid.dropna(subset=[meta_lat_col, meta_lon_col], inplace=True)
        if df_anom_calc_grid.empty: print(f"  Warn: No stations after coord conversion for {series_name}"); return pd.Series(index=all_years_index, dtype=float)
        df_anom_calc_grid['lat_bin'] = np.floor((df_anom_calc_grid[meta_lat_col] - lat_min) / grid_res).astype(int)
        df_anom_calc_grid['lon_bin'] = np.floor((df_anom_calc_grid[meta_lon_col] - lon_min) / grid_res).astype(int)
        num_lat_bins_calc = int(np.ceil((lat_max - lat_min) / grid_res)); num_lon_bins_calc = int(np.ceil((lon_max - lon_min) / grid_res))
        df_anom_calc_grid['lat_bin'] = df_anom_calc_grid['lat_bin'].clip(0, num_lat_bins_calc - 1); df_anom_calc_grid['lon_bin'] = df_anom_calc_grid['lon_bin'].clip(0, num_lon_bins_calc - 1)
        grid_cell_means = df_anom_calc_grid.groupby(['year', 'lat_bin', 'lon_bin'])['anomaly'].mean().reset_index()
        grid_cell_means['lat_center'] = lat_min + (grid_cell_means['lat_bin'] + 0.5) * grid_res; grid_cell_means['lat_center'] = grid_cell_means['lat_center'].clip(lat_min, lat_max)
        grid_cell_means['weight'] = np.cos(np.deg2rad(grid_cell_means['lat_center'])).clip(0)
        grid_cell_means = grid_cell_means[grid_cell_means['weight'] > 1e-9]
        if grid_cell_means.empty: print(f"  Warn: No grid cells with valid weight for {series_name}"); return pd.Series(index=all_years_index, dtype=float)
        grouped_data = grid_cell_means.groupby('year')
        weighted_yearly_means = grouped_data.apply(lambda g: np.average(g['anomaly'], weights=g['weight']) if g['weight'].sum() > 1e-9 else np.nan, include_groups=False)
        gridded_anomaly_series = pd.Series(index=all_years_index, dtype=float); gridded_anomaly_series.update(weighted_yearly_means)
        gridded_anomaly_series.replace([np.inf, -np.inf], np.nan, inplace=True)
        print(f"  Finished gridded anomaly calculation for {series_name}.")
        return gridded_anomaly_series

def apply_loess(series, frac, name):
    if not lowess_available: print(f"  Skip LOESS {name}: lowess NA."); return None
    if series is None or series.dropna().empty: print(f"  Skip LOESS {name}: No data."); return None
    series_clean = series.dropna(); y_smooth = series_clean.values; x_smooth = series_clean.index.values
    sort_order = np.argsort(x_smooth); x_smooth = x_smooth[sort_order]; y_smooth = y_smooth[sort_order]
    if len(y_smooth) < 10: print(f"  Skip LOESS {name}: {len(y_smooth)} pts < 10."); return None

        print(f"  LOESS applied ({name}, frac={frac*100:.0f}%).")
        return smoothed_full

    *   **BU Filter:** The accuracy of the "USA BU Filtered" series depends on the external BU metadata file, the chosen column (`Built_2020_10km_percent`), and the threshold (0.1-2.0%). It uses 2020 conditions for the entire period.

**2. Gridding Calculation (Global & USA BU Filtered Grid):**

*   **Method:**
    1.  Use station anomalies calculated as above.
    2.  Assign stations to 5x5 degree grid cells.
    3.  Average anomalies within each grid cell per year.
    4.  Calculate a latitude-weighted average (cosine weighting) of all grid cell anomalies for each year.
*   **Confidence:** Medium to High on the *implementation* of this standard method.
*   **Caveats:**
    *   **Station Coverage Over Time:** **Major caveat.** Global coverage was very sparse and biased pre-1950s. Early global averages are less reliable than recent ones. USA BU Gridded series is also affected by changing station density over time within the US.
    *   **Grid Resolution (5x5):** Coarse resolution, necessary due to data sparsity, smooths local details.
    *   **Within-Cell Averaging:** Simple mean within cells; no complex spatial interpolation.
    *   **Representativeness:** Grid cell values are large-area averages.

**In Summary:**

*   The **anomaly calculation method** is standard. Baseline choice is key for absolute values.
*   **Simple regional averages** are good indicators but sensitive to station network changes.
*   **Gridding uses standard techniques**, but reliability (especially global) is lower in early decades due to data sparsity. The "USA BU Filtered (GRIDDED)" series is calculated using this method. The non-gridded "USA BU Filtered (Station Avg)" is calculated using simple averaging of anomalies.
*   **BU filtering** depends on external data quality and filter choices.

The plot effectively shows *relative trends* and *differences* between series based on GHCNv4, but interpret early global values cautiously.
